abstract class Character
	attributes:
	  - int x
	  - int y
	  - int weaponStrength
	  - int weaponType ?
	  - int livesLeft
	methods:
	  - void Draw
	  - void isHit
	  - void Fire
	  - void Move
	  - void Die

class Hero : Character
	attributes:
	  - int x
	  - int y
	  - int weaponStrength
	  - int weaponType ?
	  - int livesLeft
	  - float score

	methods:
	  - void Draw
	    // Circle area + custom appearance
	  - void isHit
	    // collision with enemies " bullet "
	  - void Fire
	    // shoots an Index
	  - void Move
	    //  W A S D - controling
	  - void Die
	    // custom animation and some pop up
	  - void Upgrade
	    // whenever the hero gets some Rune, he uses its advantages
